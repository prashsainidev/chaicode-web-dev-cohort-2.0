// ── 1. Elements select karo ─────────────────────────
const input     = document.getElementById('todo-input');
const addBtn    = document.getElementById('todo-add-btn');
const list      = document.getElementById('todo-list');
const statTotal = document.getElementById('stat-total');
const statDone  = document.getElementById('stat-done');
const clearBtn  = document.getElementById('clear-done');

// ── 2. Stats update function ────────────────────────
function updateStats() {
  const all  = list.querySelectorAll('.todo-item');
  const done = list.querySelectorAll('.todo-item.done');
  statTotal.textContent = all.length;
  statDone.textContent  = done.length;
}

// ── 3. Empty state dikhao ────────────────────────────
function renderEmpty() {
  const existing = list.querySelector('.empty-state');
  if (list.querySelectorAll('.todo-item').length === 0 && !existing) {
    const empty = document.createElement('li');
    empty.className = 'empty-state';
    empty.textContent = 'Koi task nahi — add karo upar se! 🎉';
    list.append(empty);
  } else if (list.querySelectorAll('.todo-item').length > 0 && existing) {
    existing.remove();
  }
}

// ── 4. New todo item banana ───────────────────────────
function createTodoItem(text) {
  const li = document.createElement('li');
  li.className = 'todo-item';

  // Checkmark button
  const check = document.createElement('button');
  check.className = 'todo-check';
  check.dataset.action = 'toggle';   // delegation ke liye

  // Text span
  const span = document.createElement('span');
  span.className = 'todo-text';
  span.textContent = text;             // safe textContent!

  // Delete button
  const del = document.createElement('button');
  del.className = 'todo-del';
  del.textContent = 'Delete';
  del.dataset.action = 'delete';    // delegation ke liye

  // Sab li mein daalo
  li.append(check, span, del);
  return li;
}

// ── 5. Add task ──────────────────────────────────────
function addTask() {
  const text = input.value.trim();
  if (text === '') return;

  const item = createTodoItem(text);
  list.append(item);
  input.value = '';
  input.focus();
  updateStats();
  renderEmpty();
}

// ── 6. Add button + Enter key ────────────────────────
addBtn.addEventListener('click', addTask);
input.addEventListener('keydown', function(e) {
  if (e.key === 'Enter') addTask();
});

// ── 7. Event Delegation — list pe ek listener ────────
list.addEventListener('click', function(e) {
  const btn = e.target.closest('[data-action]'); // koi bhi data-action wala
  if (!btn) return;

  const action = btn.dataset.action;
  const li     = btn.closest('.todo-item');

  if (action === 'toggle') {
    li.classList.toggle('done');
    btn.textContent = li.classList.contains('done') ? '✓' : '';
  }

  if (action === 'delete') {
    li.remove();
    renderEmpty();
  }

  updateStats();
});

// ── 8. Filter buttons ────────────────────────────────
document.querySelectorAll('.filter-btn').forEach(function(btn) {
  btn.addEventListener('click', function() {
    // Active class update
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');

    const filter = btn.dataset.filter;

    list.querySelectorAll('.todo-item').forEach(function(item) {
      if (filter === 'all') {
        item.style.display = '';
      } else if (filter === 'done') {
        item.style.display = item.classList.contains('done') ? '' : 'none';
      } else if (filter === 'pending') {
        item.style.display = !item.classList.contains('done') ? '' : 'none';
      }
    });
  });
});

// ── 9. Clear done items ──────────────────────────────
clearBtn.addEventListener('click', function() {
  list.querySelectorAll('.todo-item.done').forEach(item => item.remove());
  updateStats();
  renderEmpty();
});

// ── Initial state ────────────────────────────────────
renderEmpty();
updateStats();
