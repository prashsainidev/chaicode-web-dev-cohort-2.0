TASK 1

addEventListener — Basics

  1. demo-btn pe click listener lagao — console mein "Button click hua!" print karo
  2. Agle click pe event object print karo — e.type, e.target dekho
  3. demo-input pe input event lagao — har keypress pe value print ho
📄 JavaScriptTASK 1
// ── TASK 1: addEventListener ─────────────────────────
const btn   = document.getElementById('demo-btn');
const input = document.getElementById('demo-input');

// 1. Click listener
btn.addEventListener('click', function() {
  console.log('Button click hua!');
});

// 2. Event object explore karo
btn.addEventListener('click', function(e) {   // 'e' = event object
  console.log('Event type:',   e.type);    // 'click'
  console.log('Target element:', e.target);  // button element
  console.log('Mouse X:',       e.clientX); // cursor ki position
});

// 3. Input event — har change pe fire hota hai
input.addEventListener('input', function(e) {
  console.log('Current value:', e./* target.value */);
});
element.addEventListener('event-name', callback)
Callback mein pehla parameter = event object (e, event, whatever naam do)
e.target = jis element pe event hua
e.target.value = input ka current value
e.type = event ka type ('click', 'input', etc.)
TASK 2

Mouse Events — Hover Effects

  1. hover-box pe mouseenter event — class hovered add karo
  2. hover-box pe mouseleave event — class hovered remove karo
  3. mouseenter ke baad text change karo: "Mouse andar hai!"
  4. mouseleave ke baad text wapas original karo
📄 JavaScriptTASK 2
// ── TASK 2: Mouse Events ─────────────────────────────
const hoverBox = document.getElementById('hover-box');
const originalText = hoverBox.textContent;

hoverBox.addEventListener('mouseenter', function() {
  hoverBox.classList.add('hovered');
  hoverBox.textContent = 'Mouse andar hai! 🎯';
});

hoverBox.addEventListener(/* 'mouseleave' */, function() {
  hoverBox.classList./* remove */('hovered');
  hoverBox./* textContent */ = originalText;
});
TASK 3

Keyboard Events — Enter Press Detect Karo

  1. demo-input pe keydown event lagao
  2. Agar Enter key press ho (e.key === 'Enter') toh alert dikhao ya console mein print karo
  3. Agar Escape press ho toh input clear karo
📄 JavaScriptTASK 3
// ── TASK 3: Keyboard Events ──────────────────────────
const input = document.getElementById('demo-input');

input.addEventListener('keydown', function(e) {
  console.log('Key pressed:', e.key); // exact key naam milta hai

  // Enter check
  if (e.key === 'Enter') {
    console.log('Enter press hua! Value:', e.target.value);
  }

  // Escape — input clear karo
  if (e.key === /* 'Escape' */) {
    e.target.value = '';
  }
});
TASK 4

Form Validation — preventDefault ka Real Use

Neeche ek form hai. Submit hone par page reload nahi hona chahiye — preventDefault() use karo. Validation bhi karo — empty fields submit na ho sakein.

📄 JavaScriptTASK 4 — Form Validation 
// ── TASK 4: Form + preventDefault ───────────────────
const submitBtn  = document.getElementById('submit-btn');
const nameField  = document.getElementById('name-field');
const emailField = document.getElementById('email-field');
const result     = document.getElementById('form-result');

submitBtn.addEventListener('click', function(e) {
  // 1. Page reload rokna — IMPORTANT!
  // (yahan form tag nahi hai, but habit mein rakho)

  let valid = true;

  // 2. Name validation
  if (nameField.value.trim() === '') {
    showError('name-msg', 'name-field', 'Naam zaroor bharo!');
    valid = false;
  } else {
    showSuccess('name-msg', 'name-field', '✓ Sahi hai');
  }

  // 3. Email validation
  if (!emailField.value.includes('@')) {
    showError('email-msg', 'email-field', 'Valid email chahiye!');
    valid = false;
  } else {
    showSuccess('email-msg', 'email-field', '✓ Sahi hai');
  }

  // 4. Sab valid ho toh success message
  if (valid) {
    result.textContent = '✅ Form submit ho gaya! Name: ' + nameField.value;
    result.style.color = '#3ddc97';
  }
});

// Helper functions
function showError(msgId, fieldId, msg) {
  const m = document.getElementById(msgId);
  m.textContent = msg; m.className = 'validation-msg show err';
  document.getElementById(fieldId).className = 'error';
}
function showSuccess(msgId, fieldId, msg) {
  const m = document.getElementById(msgId);
  m.textContent = msg; m.className = 'validation-msg show ok';
  document.getElementById(fieldId).className = 'success';
}
